CREATE TABLE person (
  id int unsigned NOT NULL AUTO_INCREMENT UNIQUE COMMENT 'Идентификатор строки',
  first_name varchar(100)  NOT NULL COMMENT 'Имя',
  last_name varchar(100)  NOT NULL COMMENT 'Фамилия',
  second_name varchar(100)  NOT NULL COMMENT 'Отчество',
  gender char(1)  NOT NULL COMMENT 'Пол',
  birthday date NOT NULL COMMENT 'Дата рождения',
  inn int unsigned NOT NULL COMMENT 'ИНН',
  country_id int NOT NULL COMMENT 'Ссылка на страну/гражданство',
  place_birth varchar(100)  DEFAULT NULL COMMENT 'Место рождения',
  resident bool DEFAULT TRUE COMMENT 'Налоговый резидент',
  marital_status_id int DEFAULT NULL COMMENT 'Ссылка семейное положение',
  education_id int DEFAULT NULL COMMENT 'Ссылка образование',
  created_at datetime DEFAULT CURRENT_TIMESTAMP COMMENT 'Время создания строки',
  updated_at datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT 'Время обновления строки',
  PRIMARY KEY (id),
  UNIQUE KEY inn (inn)
) COMMENT='физ лицо';

CREATE TABLE positions (
